			limit = {
				has_province_modifier = edict_of_zetulii
			}
			remove_province_modifier = edict_of_zetulii
		}
	}
	on_invalid = {
		custom_tooltip = kolmejar_on_removed_suomi_provinces
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = edict_of_zetulii
			}
			remove_province_modifier = edict_of_zetulii
		}

	}

	#}
