custom_button = {
	name = AT_button2
	potential = {
		has_country_flag = atm_shop_flag
	}
	trigger = {
		army_tradition = 10
		check_variable = {
			which = war_lesson_counter
			value = 5
		}
	}
	effect = {
		add_army_tradition = -10
		subtract_variable = {
			which = war_lesson_counter
			value = 5
		}
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = fire_damage_tt_var
						value = 5
					}
				}
			}
			add_country_modifier = {
				name = fire_damage_lv_1
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				check_variable = {
					which = fire_damage_tt_var
					value = 5
				}
				NOT = {
					check_variable = {
						which = fire_damage_tt_var
						value = 10
					}
				}
			}
			remove_country_modifier = fire_damage_lv_1
			add_country_modifier = {
				name = fire_damage_lv_2
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				check_variable = {
					which = fire_damage_tt_var
					value = 10
				}
				NOT = {
					check_variable = {
						which = fire_damage_tt_var
						value = 15
					}
				}
			}
			remove_country_modifier = fire_damage_lv_2
			add_country_modifier = {
				name = fire_damage_lv_3
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				check_variable = {
					which = fire_damage_tt_var
					value = 15
				}
				NOT = {
					check_variable = {
						which = fire_damage_tt_var
						value = 20
					}
				}
			}
			remove_country_modifier = fire_damage_lv_3
			add_country_modifier = {
				name = fire_damage_lv_4
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				check_variable = {
					which = fire_damage_tt_var
					value = 20
				}
			}
			remove_country_modifier = fire_damage_lv_4
			add_country_modifier = {
				name = fire_damage_lv_5
				duration = -1
				hidden = yes
			}
		}
		if = {
			limit = {
				check_variable = {
					which = fire_damage_tt_var
					value = 20
				}
			}
			remove_country_modifier = fire_damage_lv_4
			add_country_modifier = {
				name = fire_damage_lv_5
				duration = -1
				hidden = yes
			}
		}
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = fire_damage_tt_var
							value = 20
						}
					}
				}
				change_variable = {
					which = fire_damage_tt_var
					value = 5
				}
			}
		}
		#play_sound = get_war_declare
	}
	tooltip = AT_button2_tt
}
