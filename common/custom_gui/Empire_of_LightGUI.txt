	name = eol_button_1
	potential = {
		has_country_flag = open_eol_menu_flag
		NOT = {
			has_global_flag = decree_eol_1_done
		}
	}
	trigger = {
		tag = LE0
		prestige = 20
		if = {
			limit = {
				has_institution = renaissance
			}
			adm_power = 500
			dip_power = 500 
			mil_power = 500
		}else = {
			adm_power = 1000
			dip_power = 1000 
			mil_power = 1000
		}
	}
	effect = {
		if = {
			limit = {
				has_institution = renaissance
			}
			add_adm_power = -500
			add_dip_power = -500 
			add_mil_power = -500
		}else = {
			add_adm_power = -1000
			add_dip_power = -1000
			add_mil_power = -1000
		}
		set_global_flag = decree_eol_1_done
		custom_tooltip = get_cool_modifier_from_light_vassals
	}
	tooltip = eol_have_institution_1
}
