custom_button = { 
	name = experiment_1_button
	potential = {
		has_country_flag = mech_ongoing_experiment
		NOT = {
			has_country_flag = complete_1_experiment
		}
	}
	trigger = {
		if = {
			limit = {
				has_country_flag = experiment_demand_1a_flag
			}
			ROOT = {
				any_owned_province = {
					has_building = workshop
				}
			}
		}
		if = {
			limit = {
				has_country_flag = experiment_demand_1b_flag
			}
			ROOT = {
				has_adm_advisor_3 = yes
			}
		}
		if = {
			limit = {
				has_country_flag = experiment_demand_1c_flag
			}
			any_province = {
				num_of_units_in_province = {
					amount = 5
					type = infantry
				}
				trade_goods = iron
			}
		}
	}
	effect = {	
		set_country_flag = complete_1_experiment
	}
	tooltip = experiment_1_button_text
}
