
help_cathkin_action = {
	category = relation

	is_visible = {
		has_government_mechanic = saints_and_sinners
    }
	require_acceptance = no
	is_allowed = {
		curia_treasury_size = 500
		is_at_war = no
		FROM = {
			religion = catholic
			at_war_with_religious_enemy = yes
		}
    }
	on_accept = {
		FROM = {
			add_treasury = 500
			add_trust = {
				who = ROOT
				value = 10
			}
		}
		ROOT = {
			add_curia_treasury = -500
			add_5_p_a = yes
		}
    }

	ai_will_do = {
		always = yes
    }
}

papal_vassalization_action = {

    }
}

papal_integrate_action = {
	category = influence

	is_visible = {
		OR = {
			tag = PAP
			was_tag = PAP
		}
		if = {
			limit = {
				has_country_flag = easy_integration_pap_flag
			}
			OR = {
				FROM = {
					is_subject_of = PAP
				}
				FROM = {
					is_subject_of = PA0
				}
			}
		}else = {
			FROM = {
				is_subject_of_type = papal_vassal
			}
		}
    }
	is_allowed = {
		is_at_war = no
		FROM = {
			NOT = {
				liberty_desire = 40
			}
			has_opinion = {
				who = ROOT
				value = 150
			}
			years_in_vassalage_under = {
				who = ROOT
				years = 10
			}
		}
    }
	
	on_accept = {
		ROOT = {
			if = {
				limit = {
					has_government_mechanic = saints_and_sinners
				}
				lose_15_p_a = yes
			}else = {
				add_devotion = -25
			}
			inherit = FROM
		}
    }

	ai_will_do = {
		always = yes
    }
}


pontifex_expansion_action = {
	category = influence

	is_visible = {
		tag = PA0
		hre_leagues_enabled = yes
		FROM = {
			is_subject = no
			religion = catholic
		}
    }
	is_allowed = {
		is_at_war = no
		FROM = {
			is_emperor = no
			is_subject = no
			religion = catholic
			has_opinion = {
				who = ROOT
				value = 180
			}
		}
    }
	
	on_accept = {
		FROM = {
			join_league = pontifex
			change_religion = pontifex
			hidden_effect = {
				every_owned_province = {
					limit = {
						religion = catholic
					}
					change_religion = pontifex
				}
				every_subject_country = {
					limit = {
						religion = catholic
					}
					change_religion = pontifex
					every_owned_province = {
						limit = {
							religion = catholic
						}
						change_religion = pontifex
					}
				}
			}
		}
    }

	ai_will_do = {
		always = yes
    }
}

			limit = {

				}
			}

		}
		if = {
			limit = {
				NOT = { has_country_flag = exported_grain_in_oceania }

				}
				
			}

		}

				NOT = { has_country_flag = exported_grain_in_africa }
				FROM = {

						continent = africa

				}
			}
			set_country_flag = exported_grain_in_africa
