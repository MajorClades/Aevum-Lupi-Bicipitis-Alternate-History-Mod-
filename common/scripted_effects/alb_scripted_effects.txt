upgrade_vineyard = {
	if = {
		limit = {
			NOT = {
				OR = {
					has_province_modifier = vineyard_lvl_1
					has_province_modifier = vineyard_lvl_2
					has_province_modifier = vineyard_lvl_3
					has_province_modifier = vineyard_lvl_4
				}
			}
		}
		add_province_modifier = {
			name = vineyard_lvl_1
			duration = -1
			hidden = yes
		}
		owner = {
			add_treasury = -250
		}
	}
	else_if = {
		limit = {
			has_province_modifier = vineyard_lvl_1
		}
		add_province_modifier = {
			name = vineyard_lvl_2
			duration = -1
			hidden = yes
		}
		remove_province_modifier = vineyard_lvl_1
		owner = {
			add_treasury = -500
		}
	}
	else_if = {
		limit = {
			has_province_modifier = vineyard_lvl_2
		}
		add_province_modifier = {
			name = vineyard_lvl_3
			duration = -1
			hidden = yes
		}
			remove_province_modifier = vineyard_lvl_2
		owner = {
			add_treasury = -750
		}
	}
	else_if = {
		limit = {
			has_province_modifier = vineyard_lvl_3
		}
		add_province_modifier = {
			name = vineyard_lvl_4
			duration = -1
			hidden = yes
		}
			remove_province_modifier = vineyard_lvl_3
		owner = {
			add_treasury = -1000
		}
	}
}

add_or_upgrade_mandala = {
	if = {
		limit = {
			FROM = { 
				NOT = {
					OR = {
						has_province_modifier = small_size_mandala
						has_province_modifier = mid_size_mandala
						has_province_modifier = large_size_mandala
						has_province_modifier = largest_size_mandala
					}
				}
			}
		}
		add_province_modifier = {
			name = small_size_mandala
			duration = -1
		}
	}
	else_if = {
		limit = {
			FROM = { has_province_modifier = small_size_mandala }
		}
		remove_province_modifier = small_size_mandala
			add_province_modifier = {
				name = mid_size_mandala
				duration = -1
			}
	}
	else_if = {
		limit = {
			FROM = { has_province_modifier = mid_size_mandala }
		}
		remove_province_modifier = mid_size_mandala
		add_province_modifier = {
			name = large_size_mandala
			duration = -1
		}
	}
	else_if = {
		limit = {
			FROM = { has_province_modifier = large_size_mandala }
		}
		remove_province_modifier = large_size_mandala
		add_province_modifier = {
			name = largest_size_mandala
			duration = -1
		}
	}
	custom_tooltip = alb_modifier_move_to_new_capital_tt
}
