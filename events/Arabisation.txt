namespace = arabisation

country_event = { #crescent
    id = arabisation.1
    title = arabisation.1.t
    desc = arabisation.1.d
    picture = BIG_ARABIAN_CITY_eventPicture

    trigger = {
        any_owned_province = {
            OR = {
                culture_group = aramaic_group
                culture = sirian
                culture = persian
            }
            religion = rashidi
            OR = {
                region = mashriq_region
                region = mesopotamia_region
            }
        }
    }
    mean_time_to_happen = {
        months = 60
    }
    option = {
        name = arabisation.1.a
        random_owned_province = {
            limit = {
                OR = {
                    culture_group = aramaic_group
                    culture = sirian
                    culture = persian
                }
                religion = rashidi
                OR = {
                    region = mashriq_region
                    region = mesopotamia_region
                }
                NOT = { culture = crescent_arabic }
            }
            change_culture = crescent_arabic
        }
        random_owned_province = {
            limit = {
                OR = {
                    culture_group = aramaic_group
                    culture = sirian
                    culture = persian
                }
                religion = rashidi
                OR = {
                    region = mashriq_region
                    region = mesopotamia_region
                }
                NOT = { culture = crescent_arabic }
            }
            change_culture = crescent_arabic
        }
    }
}
