country_event = { 
	id = scotii.4
	title = scotii.4.test
	desc = scotii.4.desc
	picture = KINGS_CHOICE_eventPicture
	fire_only_once = yes
	trigger = {
		has_government_mechanic = scot_clans_mechanic
		NOT = {
			has_government_power = { mechanic_type = scot_clans_mechanic power_type = scot_single_power value = 30 } 
		}

	}
	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = scotii.4a
		add_government_power = { mechanic_type = scot_clans_mechanic power_type = scot_single_power value = -10 } 
		if = {
			limit = {
				owns = 4119
			}
			4119 = {
				spawn_rebels = {
					size = 3
					type = pretender_rebels
					win = yes 
					leader = "Angus"
					leader_dynasty = "McFife"
				}
			}
		}else = {
			random_owned_province = {
				limit = {
					culture = scotian
				}
				spawn_rebels = {
					size = 3
					type = pretender_rebels
					win = yes 
					leader = "Angus"
					leader_dynasty = "McFife"
				}
			}
		}
		add_ruler_modifier = {
			name = sco_loyal_few
			duration = 10950
		}
	}

	option = {
		name = scotii.4b
		add_government_power = { mechanic_type = scot_clans_mechanic power_type = scot_single_power value = 30 } 
		define_ruler = {
			name = "Angus"
			dynasty = "McFife"
			adm = 2
			dip = 4
			mil = 6
		}
		define_ruler_to_general = {
			fire = 3
			shock = 6
			manuever = 3
			siege = 3
		}
		hidden_effect = {
			kill_consort = yes
			kill_heir = {
				allow_new_heir = no
			}
		}
	}	
}
