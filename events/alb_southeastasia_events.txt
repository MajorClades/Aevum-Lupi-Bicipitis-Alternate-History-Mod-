namespace = alb_sea

#Deal with the Sem
country_event = {
	id = alb_sea.1
	title = alb_sea.1.test
	desc = alb_sea.1.desc
	picture = KINGS_CHOICE_eventPicture
	is_triggered_only = yes
	
	fire_only_once = yes
	option = {
		name = alb_sea.1.a #Mueang Yang Supremacy
		if = {
			limit = {
				TH6 = {
					is_subject_of = ROOT
				}
			}
			every_owned_province = {
				limit = {
					area = sukhothai_area
				}
				cede_province = TH6
			}
			TH6 = {
				change_religion = buddhism
				khorat_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				champasak_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				vientiane_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				north_laos_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
			}
			while = {
				add_adm_power = 50
			}
			custom_tooltip = mon_returned_50_adm_tooltip
		}
		else = {
			every_owned_province = {
				limit = {
					area = sukhothai_area
				}
				cede_province = TH6
			}
			create_march = TH6
			TH6 = {
				change_religion = buddhism
				add_country_modifier = {
					name = sem_march
					duration = -1
				}
				add_liberty_desire = -20
				khorat_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				champasak_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				vientiane_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
				north_laos_area = {
					limit = { NOT = { is_core = PREV is_permanent_claim = PREV } }
					add_permanent_claim = PREV
				}
			}
			while = {
				add_adm_power = 50
			}
			custom_tooltip = mon_freed_sem
			custom_tooltip = mon_returned_50_adm_tooltip
		}
	}
	
	option = {
		name = alb_sea.1.b #Become Sem
		change_primary_culture = central_thai

		every_owned_province = {
				limit = {
					culture = mon
				}
				add_province_modifier = {
					name = mon_mad_MON
					duration = 9125
				}
			}
	}
	
	option = {
		name = alb_sea.1.c #Purge Sem

		add_stability = -1
		
		every_owned_province = {
				limit = {
					culture = central_thai 
				}
				add_province_modifier = {
					name = sem_purge_MON
					duration = 5475
				}
			}

		2389 = {
			spawn_rebels = {
				type = nationalist_rebels
				size = 2
			}
		}
	}
	
}
