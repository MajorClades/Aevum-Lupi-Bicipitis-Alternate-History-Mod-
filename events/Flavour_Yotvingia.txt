country_event = {
    id = yotvingia.2
    title = yotvingia.2.title
    desc = yotvingia.2.desc
    picture = REFUGEES_ESCAPING_eventPicture
    trigger = {
        is_tag = LT7
        has_country_flag = { yot_genocide_saxons }
        any_owned_province = {
            OR = {
                culture = saxonard
                culture = balten_new
                culture = elutsian
            }
        }
    }
    mean_time_to_happen = {
        months = 60
    }
    option = {
        name = yotvingia.2.a
        random_owned_province = {
                limit = {
                    OR = {
                        culture = saxonard
                        culture = balten_new
                        culture = elutsian
                    }
                }
                if ={
                    limit = {
                        OR = {
                            culture = saxonard
                            AND = {
                                culture = balten_new
                                area = mazovia_area
                            }
                        }
                    }
                    change_culture = yotvingian    
                }
                else_if = {
                    limit = {
                        culture = balten_new
                        NOT = {
                            area = mazovia_area
                        }
                    }
                    change_culture = curonian
                }
                else = {
                    limit = {
                        culture = elutsian
                    }
                    change_culture = balt_elutsian
                } 
                add_base_tax = -1
                add_base_production = -1
                add_base_manpower = -1
                add_devastation = 70
        }
        random_owned_province = {
                limit = {

                }
