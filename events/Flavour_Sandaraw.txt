country_event = { #Oracle of Siwa Offers Aid
	id = sand.1
	title = sand.1.test
	desc = sand.1.desc
	picture = muslimgfx_DIPLOMACY_eventPicture
	
	trigger = {
		tag = RU0
		num_of_owned_provinces_with = {
			value = 10
			owned_by = ROOT
			controlled_by = AF6
		}
	}
	mean_time_to_happen = {
		days = 15
	}
	fire_only_once = yes
	option = { # Do they think us incompetent!?
		name = sand.1a
		ai_chance = { factor = 90 }
		add_prestige = 7
	}

	option = { # It's their lives they're throwing away...
		name = sand.1b
		ai_chance = { factor = 10 }
		every_province = {
			limit = { area = al_wahat_area }
			add_core = SDW
			change_religion = manich
		}
		release = SDW
		SDW = {
			change_government = theocracy
			add_country_modifier = {
				name = sand_countercrusade_ai
				duration = -1
			}
			create_unit_forcelimit_percentage = {
				type = infantry
				force_limit_percentage = 0.5
			}
			join_all_defensive_wars_of = ROOT
			every_owned_province = { remove_core = RU0 }
			set_consort_flag = sand_counter
		}
	}

	option = { # Only if I may lead their forces myself.
		name = sand.1c
		trigger = { ai = no }
		every_province = {
			limit = { area = al_wahat_area }
			add_core = SDW
			change_religion = manich
		}
		release = SDW
		SDW = {
			change_government = theocracy
			add_country_modifier = {
				name = sand_countercrusade
				duration = -1
			}
			create_unit_forcelimit_percentage = {
				type = infantry
				force_limit_percentage = 0.5
			}
			join_all_defensive_wars_of = ROOT
			every_owned_province = { remove_core = RU0 }
			set_country_flag = sand_counter
		}
		switch_tag = SDW
	}
}
