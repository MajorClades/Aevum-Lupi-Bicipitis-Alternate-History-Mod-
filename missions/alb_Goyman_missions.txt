	potential = {

			tag = MD1

				was_tag = MD1
			}
		}
	}

	MD1_war_of_resistance = {
		icon = mam_hormuz_straits
		position = 1

		trigger = {
			is_at_war = yes
			OR = {
				num_of_estate_privileges = {
					estate = estate_nobles
					value = 3
				}
				is_subject = yes
			}
		}

		effect = {
			add_country_modifier = {
				name = goy_war_of_fire_start
				duration = 3650
			}
		}
	}

	MD1_victory_for_god = {
		icon = qara_qoyunlu_dominate_the_gulf
		position = 3
		required_missions = { MD1_war_of_resistance } 

		trigger = {
			is_at_war = no
			grown_by_development = 25
		}

		effect = {
			persia_region = {
				add_permanent_claim = ROOT
			}
			khorasan_region = {
				add_permanent_claim = ROOT
			}
		}
	}

	MD1_iran_proper = {
		icon = mission_russian_patriarchatee
		position = 4
		required_missions = { MD1_victory_for_god } 
		provinces_to_highlight = {
			region = persia_region
			NOT = { country_or_non_sovereign_subject_holds = ROOT } 
		}

		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 25
				region = persia_region
			}
		}

		effect = {
			add_yearly_manpower = 1.5
			capital_scope = {
				add_or_upgrade_manpower_building = yes
				add_or_upgrade_fort_building = yes
			}
		}
	}
	MD1_strenght_of_god = {
		icon = mission_russian_patriarchatee
		position = 5
		required_missions = { MD1_iran_proper MD1_khorasan_region} 
		provinces_to_highlight = {
			OR = {
				region = persia_region
				region = khorasan_region
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT } 
		}

		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 65
				religion = ROOT
				OR = {
					region = persia_region
					region = khorasan_region
				}
			}
		}

		effect = {
			capital_scope = {
				add_permanent_province_modifier = {
					name = goy_bastion_of_faith_province
					duration = -1
				}
			}
			add_country_modifier = {
				name = goy_bastion_of_faith
				duration = -1
			}
		}
	}
