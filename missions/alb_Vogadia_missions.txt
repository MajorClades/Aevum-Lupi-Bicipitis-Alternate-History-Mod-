				}

				custom_tooltip = VOGADIA_TT_5c

					name = "vog_pray_hands"
					duration = -1
				}
			}
			hidden_effect = {

					limit = { government = monarchy }

						name = "vog_king_hands"
						duration = -1
					}

					limit = { government = republic }

						name = "vog_work_hands"

					}
